1 var Resource = require('./resource');
  2 
  3 /**
  4  * Represents an account in the spire api.
  5  *
  6  * @constructor
  7  * @extends Resource
  8  * @param spire {object} Spire object
  9  * @param data {object} Account data from the spire api
 10  */
 11 function Account (spire, data) {
 12   this.spire = spire;
 13   this.data = data;
 14   this.resourceName = 'account';
 15 };
 16 
 17 Account.prototype = new Resource();
 18 
 19 module.exports = Account;
 20 
 21 /**
 22  * Updates the billing plan for the account.
 23  *
 24  * @param {object} info New billing plan data
 25  * @param {function (err, account)} cb Callback
 26  */
 27 Account.prototype.updateBillingSubscription = function (info, cb) {
 28   var account = this;
 29   this.request('update_billing_subscription', info, function (err, data) {
 30     if (err) return cb(err);
 31     account.data = data;
 32     cb(account);
 33   });
 34 };
 35 
 36 /**
 37  * Requests
 38  * These define API calls and have no side effects.  They can be run by calling
 39  *     this.request(<request name>);
 40  */
 41 
 42 /**
 43  * Updates the billing subscription with the given data.
 44  * @name update_billing_subscription
 45  * @ignore
 46  */
 47 Resource.defineRequest(Account.prototype, 'update_billing_subscription', function (info) {
 48   var billing = this.data.billing;
 49   return {
 50     method: 'put',
 51     url: billing.url,
 52     content: info,
 53     headers: {
 54       'Accept': this.mediaType(),
 55       'Content-Type': this.mediaType(),
 56       'Authorization': this.authorization(invioces)
 57     }
 58   };
 59 });

 61 /**
 62  * Gets the billing invoices for the account.
 63  * @name billing_invoices
 64  * @ignore
 65  66  */
 67 Resource.defineRequest(Account.prototype, 'billing_invoices', function () {
 68   var invoices = this.data.billing.invoices;
 69   return {
 70     method: 'get',
 71     url: invoices.url,
 72     headers: {
 73       'Accept': "application/json",
 74       'Authorization': "Capability " + invoices.capability
 75     }
 76   };
 77 });
