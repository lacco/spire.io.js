 23 /**
 24  * Spire API Client
 25  *
 26  * @class <strong>Spire API Client</strong>
 27  *
 28  * @example
 29  * var spire = new Spire();
 30  *
 31  * @constructor
 32  * @param {object} opts Options for Spire
 33  * @param {string} opts.url Spire url do use (defaults to 'https://api.spire.io')
 34  * @param {string} opts.version Version of Spire api to use (defaults to '1.0')
 35  * @param {number} opts.timeout Timeout for requests (defaults to 30 seconds)
 36  */
 37 var Spire = function (opts) {
 38   this.api = new API(this, opts);
 39   this.session = null;
 40 };
 41

/**
 45  * Get the account key.
 46  *
 47  * @returns {string} Account key
 48  */
 49 Spire.prototype.key = function () {
 50   if (this.session && this.session.resources && this.session.resources.account) {
 51     return this.session.resources.account.key();
 52   }
 53   return null;
 54 };

 65 /**
 66  * Start the Spire session with the given account key.
 67  *
 68  * @example
 69  * var spire = new Spire();
 70  * spire.stark(your_api_key, function (err, session) {
 71  *   if (!err) {
 72  *     // You now have a spire session.
 73  *     // Start creating channels and subscripions.
 74  *   }
 75  * });
 76  *
 77  * @param {string} key The acccount key
 78  * @param {function(err)} cb Callback
 79  */
 80 Spire.prototype.start = function (key, cb) {
 81   var spire = this;
 82   this.discover(function () {
 83     spire.api.createSession(key, function (err, session) {
 84       if (err) return cb(err);
 85       spire.session = session;
 86       cb(null);
 87     });
 88   });
 89 };

196 /**
197  * Gets a channel (creates if necessary).
198  *
199  * @example
200  * var spire = new Spire();
201  * spire.start(your_api_key, function (err, session) {
202  *   spire.channel('foo', function (err, channel) {
203  *     if (!err) {
204  *       // `channel` is the channel named "foo".
205  *     }
206  *   });
207  * });
208  *
209  * @param {string} name Channel name to get or create
210  * @param {function(err, channel)} cb Callback
211  */
212 Spire.prototype.channel = function (name, cb) {
213   if (this.session._channels[name]) {
214     return cb(null, this.session._channels[name]);
215   }
216   this._findOrCreateChannel(name, cb);
217 };

259 /**
260  * Gets a subscription to the given channels.  Creates the channels and the
261  * subscription if necessary.
262  *
263  * @example
264  * var spire = new Spire();
265  * spire.start(your_api_key, function (err, session) {
266  *   spire.subscribe('mySubscription', ['foo', 'bar'], function (err, subscription) {
267  *     if (!err) {
268  *       // `subscription` is a subscription named 'mySubscription', listening on channels named 'foo' and 'bar'.
269  *     }
270  *   });
271  * });
272  *
273  * @param {string} Subscription name
274  * @param {array or string} channelOrChannels Either a single channel name, or an array of
275  *   channel names to subscribe to
276  * @param {function (err, subscription)} cb Callback
277  */
278 Spire.prototype.subscribe = function (name, channelOrChannels, cb) {
279   var channelNames = (typeof channelOrChannels === 'string')
280     ? [channelOrChannels]
281     : channelOrChannels;
282 
283   var spire = this;
284   async.forEach(
285     channelNames,
286     function (channelName, innerCB) {
287       spire._findOrCreateChannel(channelName, function (err, channel) {
288         if (err) { return cb(err); }
289         innerCB();
290       });
291     },
292     function (err) {
293       if (err) return cb(err);
294       spire._findOrCreateSubscription(name, channelNames, cb);
295     }
296   );
297 };

419 /**
420  * Number of times to retry creating a channel or subscription before giving up.
421  */
422 Spire.prototype.CREATION_RETRY_LIMIT = 5;
423 
424 /**
425  * Returns the channel with name 'name', creating it if necessary.
426  * You should use `Spire.prototype.channel` method instead of this one.  This
427  * method ignores cached channels.
428  *
429  * @param {string} name Channel name
430  * @param {function (err, channel)} cb Callback
431  */
432 Spire.prototype._findOrCreateChannel = function (name, cb) {
433   var spire = this;
434   var creationCount = 0;
435 
436   var createChannel = function () {
437     creationCount++;
438     spire.session.createChannel(name, function (err, channel) {
439       if (!err) return cb(null, channel);
440       if (err.status !== 409) return cb(err);
441       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
442         return cb(new Error("Could not create channel: " + name));
443       }
444       getChannel();
445     });
446   };
447 
448   var getChannel = function () {
449     spire.session.channels$(function (err, channels) {
450       if (!err && channels[name]) return cb(null, channels[name]);
451       if (err && err.status !== 409) return cb(err);
452       createChannel();
453     });
454   };
455 
456   createChannel();
457 };
458 
459 /**
460  * Returns the subvscription with name 'name', creating it if necessary.
461  * You should use `Spire.prototype.subscribe` method instead of this one.  This
462  * method ignores cached subscriptions, and does not create the channels.
463  *
464  * @param {string} name Subscription name
465  * @param {string} channelNames Channel names
466  * @param {function (err, subscription)} cb Callback
467  */
468 Spire.prototype._findOrCreateSubscription = function (name, channelNames, cb) {
469   var spire = this;
470   var creationCount = 0;
471 
472   var createSubscription = function () {
473     creationCount++;
474     spire.session.createSubscription(name, channelNames, function (err, sub) {
475       if (!err) return cb(null, sub);
476       if (err.status !== 409) return cb(err);
477       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
478         return cb(new Error("Could not create subscription: " + name));
479       }
480       getSubscription();
481     });
482   };
483 
484   var getSubscription = function () {
485     spire.session.subscriptions$(function (err, subscriptions) {
486       if (!err && subscriptions[name]) return cb(null, subscriptions[name]);
487       if (err && err.status !== 409) return cb(err);
488       createSubscription();
489     });
490   };
491 
492   createSubscription();
493 };
