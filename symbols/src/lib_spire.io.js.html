 23 /**
 24  * Spire API Client
 25  *
 26  * @class <strong>Spire API Client</strong>
 27  *
 28  * @example
 29  * var spire = new Spire();
 30  *
 31  * @example
 32  * var spire = new Spire({
 33  *   secret: my_account_secret
 34  * });
 35  *
 36  * @constructor
 37  * @param {object} [opts] Options for Spire
 38  * @param {string} [secret] The account API secret.  If you do you not set this, you
 39  * must call one of:
 40  *   * `spire.start(secret, callback)`
 41  *   * `spire.login(email, password, callback)` or
 42  *   * `spire.register(user, callback)
 43  *   before you can start creating channels.
 44  * @param {string} [opts.url] Spire url do use (defaults to 'https://api.spire.io')
 45  * @param {string} opts.version Version of Spire api to use (defaults to '1.0')
 46  * @param {number} opts.timeout Timeout for requests (defaults to 30 seconds)
 47  */
 48 function Spire(opts) {
 49   opts = opts || {};
 50   this.api = new API(this, opts);
 51   this.session = null;
 52   this._opts_secret = opts.secret;
 53 }

 57 /**
 58  * Get the account secret.
 59  *
 60  * @returns {string} Account secret
 61  */
 62 Spire.prototype.secret = function () {
 63   this._ensureSession();
 64   if (this.session && this.session.resources && this.session.resources.account) {
 65     return this.session.resources.account.secret();
 66   }
 67   return null;
 68 };

 71  * Start the Spire session with the given account secret.

 73  * @example
 74  * var spire = new Spire();
 75  * spire.start(your_api_secret, function (err, session) {
 76  *   if (!err) {
 77  *     // You now have a spire session.
 78  *     // Start creating channels and subscripions.
 79  *   }
 80  * });
 81  *
 82  * @param {string} secret The acccount secret
 83  * @param {function(err)} cb Callback
 84  */
 85 Spire.prototype.start = function (secret, cb) {
 86   var spire = this;
 87   this.api.createSession(secret, function (err, session) {
 88     if (err) return cb(err);
 89     spire.session = session;
 90     cb(null);
 91   });

211  * @param {string} name Channel name to get or create
212  * @param {function(err, channel)} cb Callback

214 Spire.prototype.channel = function (name, cb) {
215   var spire = this;
216   this._ensureSession(function (err) {
217     if (err) return cb(err);
218 
219     if (spire.session._channels[name]) {
220       return cb(null, spire.session._channels[name]);
221     }
222     spire._findOrCreateChannel(name, cb);
223   });
224 };

274 /**
275  * Gets a subscription to the given channels.  Creates the channels and the
276  * subscription if necessary.
277  *
278  * @example
279  * var spire = new Spire();
280  * spire.start(your_api_secret, function (err, session) {
281  *   spire.subscription('mySubscription', ['foo', 'bar'], function (err, subscription) {
282  *     if (!err) {
283  *       // `subscription` is a subscription named 'mySubscription', listening on channels named 'foo' and 'bar'.
284  *     }
285  *   });
286  * });
287  *
288  * @param {string} Subscription name
289  * @param {array or string} channelOrChannels Either a single channel name, or an array of
290  *   channel names to subscribe to
291  * @param {function (err, subscription)} cb Callback
292  */
293 Spire.prototype.subscription = function (name, channelOrChannels, cb) {
294   var channelNames = (typeof channelOrChannels === 'string') ?
295     [channelOrChannels] : channelOrChannels;
296 
297   var spire = this;
298   this._ensureSession(function (err) {
299     if (err) return cb(err);
300     async.forEach(
301       channelNames,
302       function (channelName, innerCB) {
303         spire._findOrCreateChannel(channelName, innerCB);
304       },
305       function (err) {
306         if (err) return cb(err);
307         spire._findOrCreateSubscription(name, channelNames, cb);
308       }
309     );
310   });
311 };

361 /**
362  * Creates an new subscription to a channel or channels, and adds a listener.
363  *
364  * @example
365  * var spire = new Spire();
366  * spire.start(your_api_secret, function (err, session) {
367  *   spire.subscribe('myChannel', options, function (messages) {
368  *     // `messages` is array of messages sent to the channel
369  *   }, function (err) {
370  *   // `err` will be non-null if there was a problem creating the subscription.
371  * });
372  *
373  * By default this will get all events from the beginning of time.
374  * If you only want messages created from this point forward, pass { last: 'now' } in the options:
375  *
376  * @example
377  * var spire = new Spire();
378  * spire.start(your_api_secret, { last: 'now' }, function (err, session) {
379  *   spire.subscribe('myChannel', options, function (messages) {
380  *     // `messages` is array of messages sent to the channel
381  *   }, function (err) {
382  *   // `err` will be non-null if there was a problem creating the subscription.
383  * });
384  *
385  * @param {array or string} channelOrChannels Either a single channel name, or an array of
386  *   channel names to subscribe to
387  * @param {object} [options] Options to pass to the listener
388  * @param {function (messages)} listener Listener that will get called with each batch of messages
389  * @param {function (err, subscription)} [cb] Callback
390  */
391 Spire.prototype.subscribe = function (channelOrChannels, options, listener, cb) {
392   var spire = this;
393   if (typeof options === 'function') {
394     cb = listener;
395     listener = options;
396     options = {}
397   }
398 
399   cb = cb || function () {};
400 
401   var name = 'anon-' + Date.now() + '-' + Math.random();
402 
403   this._ensureSession(function (err) {
404     if (err) return cb(err);
405     spire.subscription(name, channelOrChannels, function (err, subscription) {
406       if (err) return cb(err);
407       subscription.addListener('messages', listener);
408       subscription.startListening(options);
409       process.nextTick(function () {
410         cb(null, subscription);
411       });
412     });
413   });
414 };

555 /**
556  * Number of times to retry creating a channel or subscription before giving up.
557  */
558 Spire.prototype.CREATION_RETRY_LIMIT = 5;
559 
560 /**
561  * Returns the channel with name 'name', creating it if necessary.
562  * You should use `Spire.prototype.channel` method instead of this one.  This
563  * method ignores cached channels.
564  *
565  * @param {string} name Channel name
566  * @param {function (err, channel)} cb Callback
567  */
568 Spire.prototype._findOrCreateChannel = function (name, cb) {
569   var spire = this;
570   var creationCount = 0;
571 
572   function createChannel() {
573     creationCount++;
574     spire.session.createChannel(name, function (err, channel) {
575       if (!err) return cb(null, channel);
576       if (err.status !== 409) return cb(err);
577       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
578         return cb(new Error("Could not create channel: " + name));
579       }
580       getChannel();
581     });
582   }
583 
584   function getChannel() {
585     spire.session.channelByName(name, function (err, channel) {
586       if (err && err.status !== 404) return cb(err);
587       if (channel) return cb(null, channel);
588       createChannel();
589     });
590   }
591 
592   this._ensureSession(function (err) {
593     if (err) return cb(err);
594     getChannel();
595   });
596 };
597 
598 /**
599  * Returns the subvscription with name 'name', creating it if necessary.
600  * You should use `Spire.prototype.subscribe` method instead of this one.  This
601  * method ignores cached subscriptions, and does not create the channels.
602  *
603  * @param {string} name Subscription name
604  * @param {string} channelNames Channel names
605  * @param {function (err, subscription)} cb Callback
606  */
607 Spire.prototype._findOrCreateSubscription = function (name, channelNames, cb) {
608   var spire = this;
609   var creationCount = 0;
610 
611   function createSubscription() {
612     creationCount++;
613     spire.session.createSubscription({
614       name: name,
615       channelNames: channelNames
616     }, function (err, sub) {
617       if (!err) return cb(null, sub);
618       if (err.status !== 409) return cb(err);
619       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
620         return cb(new Error("Could not create subscription: " + name));
621       }
622       getSubscription();
623     });
624   }
625 
626   function getSubscription() {
627     spire.session.subscriptionByName(name, function (err, subscription) {
628       if (err && err.status !== 404) return cb(err);
629       if (subscription) return cb(null, subscription);
630       createSubscription();
631     });
632   }
633 
634   this._ensureSession(function (err) {
635     if (err) return cb(err);
636     createSubscription();
637   });
638 };

640 function NoSessionError(message) {
641   this.name = "No Session Error";
642   this.message = message ||
643     "You need a Spire session to do that.\n" +
644     "Call one of:\n" +
645     "spire.start(callback)\n" +
646     "spire.login(email, password, callback)\n" +
647     "spire.register(user, callback)\n";
648 }
649 
650 NoSessionError.prototype = new Error();
651 NoSessionError.constructor = NoSessionError;
652 
653 Spire.prototype._ensureSession = function (cb) {
654   if (this.session) {
655     if (cb) return cb(null);
656     return;
657   }
658 
659   if (!this._opts_secret) {
660     var noSessionError = new NoSessionError();
661     if (cb) return cb(noSessionError);
662     throw noSessionError;
663   }
664 
665   if (!cb) {
666     throw new NoSessionError();
667   }
668 
669   this.start(this._opts_secret, cb);
670 };
