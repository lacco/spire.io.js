 23 /**
 24  * Spire API Client
 25  *
 26  * @class <strong>Spire API Client</strong>
 27  *
 28  * @example
 29  * var spire = new Spire();
 30  *
 31  * @example
 32  * var spire = new Spire({
 33  *   key: my_account_key
 34  * });
 35  *
 36  * @constructor
 37  * @param {object} [opts] Options for Spire
 38  * @param {string} [key] The account API key.  If you do you not set this, you
 39  * must call one of:
 40  *   * `spire.start(key, callback)`
 41  *   * `spire.login(email, password, callback)` or
 42  *   * `spire.register(user, callback)
 43  *   before you can start creating channels.
 44  * @param {string} [opts.url] Spire url do use (defaults to 'https://api.spire.io')
 45  * @param {string} opts.version Version of Spire api to use (defaults to '1.0')
 46  * @param {number} opts.timeout Timeout for requests (defaults to 30 seconds)
 47  */
 48 function Spire(opts) {
 49   this.api = new API(this, opts);
 50   this.session = null;
 51   this._opts_key = opts.key;
 52 }

 56 /**
 57  * Get the account key.
 58  *
 59  * @returns {string} Account key
 60  */
 61 Spire.prototype.key = function () {
 62   this._ensureSession();
 63   if (this.session && this.session.resources && this.session.resources.account) {
 64     return this.session.resources.account.key();
 65   }
 66   return null;
 67 };

 69 /**
 70  * Start the Spire session with the given account key.
 71  *
 72  * @example
 73  * var spire = new Spire();
 74  * spire.start(your_api_key, function (err, session) {
 75  *   if (!err) {
 76  *     // You now have a spire session.
 77  *     // Start creating channels and subscripions.
 78  *   }
 79  * });
 80  *
 81  * @param {string} key The acccount key
 82  * @param {function(err)} cb Callback
 83  */
 84 Spire.prototype.start = function (key, cb) {
 85   var spire = this;
 86   this.api.createSession(key, function (err, session) {
 87     if (err) return cb(err);
 88     spire.session = session;
 89     cb(null);
 90   });

195 /**
196  * Gets a channel (creates if necessary).
197  *
198  * @example
199  * var spire = new Spire();
200  * spire.start(your_api_key, function (err, session) {
201  *   spire.channel('foo', function (err, channel) {
202  *     if (!err) {
203  *       // `channel` is the channel named "foo".
204  *     }
205  *   });
206  * });
207  *
208  * @param {string} name Channel name to get or create
209  * @param {function(err, channel)} cb Callback
210  */
211 Spire.prototype.channel = function (name, cb) {
212   var spire = this;
213   this._ensureSession(function (err) {
214     if (err) return cb(err);
215 
216     if (spire.session._channels[name]) {
217       return cb(null, spire.session._channels[name]);
218     }
219     spire._findOrCreateChannel(name, cb);
220   });
221 };

271 /**
272  * Gets a subscription to the given channels.  Creates the channels and the
273  * subscription if necessary.
274  *
275  * @example
276  * var spire = new Spire();
277  * spire.start(your_api_key, function (err, session) {
278  *   spire.subscription('mySubscription', ['foo', 'bar'], function (err, subscription) {
279  *     if (!err) {
280  *       // `subscription` is a subscription named 'mySubscription', listening on channels named 'foo' and 'bar'.
281  *     }
282  *   });
283  * });
284  *
285  * @param {string} Subscription name
286  * @param {array or string} channelOrChannels Either a single channel name, or an array of
287  *   channel names to subscribe to
288  * @param {function (err, subscription)} cb Callback
289  */
290 Spire.prototype.subscription = function (name, channelOrChannels, cb) {
291   var channelNames = (typeof channelOrChannels === 'string') ?
292     [channelOrChannels] : channelOrChannels;
293 
294   var spire = this;
295   this._ensureSession(function (err) {
296     if (err) return cb(err);
297     async.forEach(
298       channelNames,
299       function (channelName, innerCB) {
300         spire._findOrCreateChannel(channelName, innerCB);
301       },
302       function (err) {
303         if (err) return cb(err);
304         spire._findOrCreateSubscription(name, channelNames, cb);
305       }
306     );
307   });
308 };

358 /**
359  * Creates an new subscription to a channel or channels, and adds a listener.
360  *
361  * @example
362  * var spire = new Spire();
363  * spire.start(your_api_key, function (err, session) {
364  *   spire.subscribe('myChannel', options, function (messages) {
365  *     // `messages` is array of messages sent to the channel
366  *   }, function (err) {
367  *   // `err` will be non-null if there was a problem creating the subscription.
368  * });
369  *
370  * @param {array or string} channelOrChannels Either a single channel name, or an array of
371  *   channel names to subscribe to
372  * @param {object} [options] Options to pass to the listener
373  * @param {function (messages)} listener Listener that will get called with each batch of messages
374  * @param {function (err, subscription)} [cb] Callback
375  */
376 Spire.prototype.subscribe = function (channelOrChannels, options, listener, cb) {
377   if (typeof options === 'function') {
378     cb = listener;
379     listener = options;
380   }
381 
382   this._ensureSession(cb);
383 
384   cb = cb || function () {};
385 
386   var name = 'anon-' + Date.now() + '-' + Math.random();
387 
388   this.subscription(name, channelOrChannels, function (err, subscription) {
389     if (err) return cb(err);
390     subscription.addListener('messages', listener);
391     subscription.startListening(options);
392     return cb(null, subscription);
393   });
394 };

534 /**
535  * Number of times to retry creating a channel or subscription before giving up.
536  */
537 Spire.prototype.CREATION_RETRY_LIMIT = 5;
538 
539 /**
540  * Returns the channel with name 'name', creating it if necessary.
541  * You should use `Spire.prototype.channel` method instead of this one.  This
542  * method ignores cached channels.
543  *
544  * @param {string} name Channel name
545  * @param {function (err, channel)} cb Callback
546  */
547 Spire.prototype._findOrCreateChannel = function (name, cb) {
548   if (!this.session) {
549     return cb(new Error("You must start spire before you can do that."));
550   }
551 
552   var spire = this;
553   var creationCount = 0;
554 
555   function createChannel() {
556     creationCount++;
557     spire.session.createChannel(name, function (err, channel) {
558       if (!err) return cb(null, channel);
559       if (err.status !== 409) return cb(err);
560       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
561         return cb(new Error("Could not create channel: " + name));
562       }
563       getChannel();
564     });
565   }
566 
567   function getChannel() {
568     spire.session.channels$(function (err, channels) {
569       if (err) return cb(err);
570       if (channels[name]) return cb(null, channels[name]);
571       createChannel();
572     });
573   }
574 
575   this._ensureSession(function (err) {
576     if (err) return cb(err);
577     getChannel();
578   });
579 };
580 
581 /**
582  * Returns the subvscription with name 'name', creating it if necessary.
583  * You should use `Spire.prototype.subscribe` method instead of this one.  This
584  * method ignores cached subscriptions, and does not create the channels.
585  *
586  * @param {string} name Subscription name
587  * @param {string} channelNames Channel names
588  * @param {function (err, subscription)} cb Callback
589  */
590 Spire.prototype._findOrCreateSubscription = function (name, channelNames, cb) {
591   if (!this.session) {
592     return cb(new Error("You must start spire before you can do that."));
593   }
594 
595   var spire = this;
596   var creationCount = 0;
597 
598   function createSubscription() {
599     creationCount++;
600     spire.session.createSubscription(name, channelNames, function (err, sub) {
601       if (!err) return cb(null, sub);
602       if (err.status !== 409) return cb(err);
603       if (creationCount >= spire.CREATION_RETRY_LIMIT) {
604         return cb(new Error("Could not create subscription: " + name));
605       }
606       getSubscription();
607     });
608   }
609 
610   function getSubscription() {
611     spire.session.subscriptions$(function (err, subscriptions) {
612       if (err) return cb(err);
613       if (subscriptions[name]) return cb(null, subscriptions[name]);
614       createSubscription();
615     });
616   }
617 
618   this._ensureSession(function (err) {
619     if (err) return cb(err);
620     getSubscription();
621   });
622 };

624 function NoSessionError(message) {
625   this.name = "No Session Error";
626   this.message = message ||
627     "You need a Spire session to do that.\n" +
628     "Call one of:\n" +
629     "spire.start(callback)\n" +
630     "spire.login(email, password, callback)\n" +
631     "spire.register(user, callback)\n";
632 }
633 
634 NoSessionError.prototype = new Error();
635 NoSessionError.constructor = NoSessionError;
636 
637 Spire.prototype._ensureSession = function (cb) {
638   if (this.session) {
639     if (cb) return cb(null);
640     return;
641   }
642 
643   if (!this._opts_key) {
644     var noSessionError = new NoSessionError();
645     if (cb) return cb(noSessionError);
646     throw noSessionError;
647   }
648 
649   if (!cb) {
650     throw new NoSessionError();
651   }
652 
653   this.start(this._opts_key, cb);
654 };
