 12 /**
 13  * Represents a session in the spire api.
 14  *
 15  * <p>Sessions contain other resources, like channels and subscriptions.  These
 16  * can be accessed in the <code>session.resources</code> object.</p>
 17  *
 18  * <p>One important resource inside <code>session</code> is the account resource
 19  * <code>session.account</code>, which is only given to sessions authenticated
 20  * with an email and password.  The account resource is documented in its own
 21  * class.</p>
 22  *
 23  * <p>Session objects maintain lists of channels and subscriptions.  If you call
 24  * the <code>session.channels</code> or <code>session.subscriptions</code>
 25  * methods, you will get back cached data if it exists.  Use the <code>$</code>
 26  * cache-bypass methods: <code>session.channels$</code> and
 27  * <code>session.subscriptions$</code> to get fresh data from the api.</p>
 28  *
 29  * @class Session Resource
 30  *
 31  * @constructor
 32  * @extends Resource
 33  * @param {object} spire Spire object
 34  * @param {object} data Session data from the spire api
 35  */
 36 function Session(spire, data) {
 37   this.spire = spire;
 38   this.data = data;
 39   this.resourceName = 'session';
 40 
 41   this._channels = {};
 42   this._subscriptions = {};
 43 
 44 	this._storeResources();
 45 
 46 }
 47 
 48 Session.prototype = new Resource();
 49 
 50 module.exports = Session;
 51 
 52 /**
 53  * <p>Gets the Session resource.
 54  *
 55  * @param {function (err, session)} cb Callback
 56  */
 57 Session.prototype.get = function (cb) {
 58   var session = this;
 59   this.request('get', function (err, data) {
 60     if (err) return cb(err);
 61     session.data = data;
 62     session._storeResources();
 63     cb(null, session);
 64   });
 65 };

 67 /**
 68  * Gets the account resource.  Only available to sessions that are authenticated
 69  * with an email and password.
 70  *
 71  * Returns a value from the cache, if one if available.
 72  *
 73  * @example
 74  * spire.session.account(function (err, account) {
 75  *   if (!err) {
 76  *     // `account` is account resource.
 77  *   }
 78  * });
 79  *
 80  * @param {function (err, account)} cb Callback
 81  */
 82 Session.prototype.account = function (cb) {
 83   if (this._account) return cb(null, this._account);
 84   this.account$(cb);
 85 };

 87 /**
 88  * Gets the account resource.  Only available to sessions that are authenticated
 89  * with an email and password.
 90  *
 91  * Always gets a fresh value from the api.
 92  93  *
 94  * @example
 95  * spire.session.account$(function (err, account) {
 96  *   if (!err) {
 97  *     // `account` is account resource.
 98  *   }
 99  * });
100  * @param {function (err, account)} cb Callback
101  */
102 Session.prototype.account$ = function (cb) {
103   var session = this;
104   this.request('account', function (err, account) {
105     if (err) return cb(err);
106     session._account = new Account(session.spire, account);
107     cb(null, session._account);
108   });
109 };

111 /**
112  * Resets the account resource.  Only available to sessions that are authenticated
113  * with an email and password.
114  * *
115  * @example
116  * spire.session.resetAccount(function (err, session) {
117  *   if (!err) {
118  *     // `session` is session with new account resource.
119  *   }
120  * });
121  * @param {function (err, session)} cb Callback
122  */
123 Session.prototype.resetAccount = function (cb) {
124   var session = this;
125   this._account.reset(function (err, sessionData) {
126     if (err) return cb(err);
127 		session.data = sessionData;
128 		session._storeResources();
129     cb(null, session);
130   });
131 };

152 /**
153  * Gets the channels collection.  Returns a hash of Channel resources.
154  *
155  * Returns a value from the cache, if one if available.
156  *
157  * @example
158  * spire.session.channels(function (err, channels) {
159  *   if (!err) {
160  *     // `channels` is a hash of all the account's channels
161  *   }
162  * });
163  *
164  * @param {function (err, channels)} cb Callback
165  */
166 Session.prototype.channels = function (cb) {
167   if (!_.isEmpty(this._channels)) return cb(null, this._channels);
168   this.channels$(cb);
169 };

171 /**
172  * Gets the channels collection.  Returns a hasg of Channel resources.
173  *
174  * Always gets a fresh value from the api.
175  *
176  * @example
177  * spire.session.channels$(function (err, channels) {
178  *   if (!err) {
179  *     // `channels` is a hash of all the account's channels
180  *   }
181  * });
182  *
183  * @param {function (err, channels)} cb Callback
184  */
185 Session.prototype.channels$ = function (cb) {
186   var session = this;
187   this.request('channels', function (err, channelsData) {
188     if (err) return cb(err);
189     _.each(channelsData, function (channel, name) {
190       session._memoizeChannel(new Channel(session.spire, channel));
191     });
192     cb(null, session._channels);
193   });
194 };

260 /**
261  * Creates a channel.  Returns a Channel resource.  Errors if a channel with the
262  * specified name exists.
263  *
264  * @param {string} name Channel name
265  * @param {number} limit Number of messages to keep in channel
266  * @param {function (err, channel)} cb Callback
267  */
268 Session.prototype.createChannel = function (name, limit, cb) {
269   var session = this;
270   if (!cb) {
271     cb = limit;
272     limit = null;
273   }
274 
275   var opts = {
276     name: name
277   }
278 
279   if (limit !== null) {
280     opts.limit = limit
281   }
282 
283   this.request('create_channel', opts, function (err, data) {
284     if (err) return cb(err);
285     var channel = new Channel(session.spire, data);
286     session._memoizeChannel(channel);
287     cb(null, channel);
288   });
289 };

291 /**
292  * Creates a subscription to any number of channels.  Returns a Subscription
293  * resource.  Errors if a subscription with the specified name exists.
294  *
295  * @param {object} options Options
296  * @param {string} options.name Subscription name
297  * @param {array} options.channelNames Channel names to subscribe to
298  * @param {array} options.channelUrls Channel urls to subscribe to
299  * @param {number} options.expiration Subscription expiration (ms)
300  * @param {function (err, subscription)} cb Callback
301  */
302 Session.prototype.createSubscription = function (options, cb) {
303   var name = options.name;
304   var channelNames = options.channelNames || [];
305   var channelUrls = options.channelUrls || [];
306   var expiration = options.expiration;
307 
308   var session = this;
309   this.channels(function (channels) {
310     channelUrls.push.apply(channelUrls, _.map(channelNames, function (name) {
311       return session._channels[name].url();
312     }));
313 314     session.request('create_subscription', name, channelUrls, expiration, function (err, sub) {
315       if (err) return cb(err);
316       var subscription = new Subscription(session.spire, sub);
317       session._memoizeSubscription(subscription);
318       cb(null, subscription);
319     });
320   });
321 };

323  /**

326  * @param channel {object} Channel to store

328 Session.prototype._memoizeChannel = function (channel) {

333  * Stores the subscription resource in a hash by its name.

337 Session.prototype._memoizeSubscription = function (subscription) {

341 /**
342  * Stores the resources.
343  */
344 Session.prototype._storeResources = function () {
345   var session = this;
346 	var resources = {};
347   _.each(this.data.resources, function (resource, name) {
348     // Turn the account object into an instance of Resource.
349     if (name === 'account') {
350       resource = new Account(session.spire, resource);
351       session._account = resource;
352     }
353     resources[name] = resource;
354   });
355 
356   this.resources = resources;
357 };
