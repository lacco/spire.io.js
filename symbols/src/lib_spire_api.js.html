 13 /**
 14  * Abstraction for the Spire API
 15  *
 16  * @class Collection of API methods
 17  *
 18  * @example
 19  * var api = new API(options);
 20  *
 21  * @constructor
 22  * @param {object} spire Spire object
 23  * @param {object} [opts] Options
 24  * @param {string} [opts.url] Spire api url
 25  * @param {string} [opts.version] Version of the Spire api to use
 26  * @param {string} [opts.timeout] Timeout for requests
 27  */
 28 var API = function (spire, opts) {
 29   this.spire = spire;
 30 
 31   opts = opts || {};
 32   this.url = opts.url || 'https://api.spire.io';
 33   this.version = opts.version || '1.0';
 34   this.timeout = opts.timeout || 30 * 1000;
 35 
 36   this.description = null;
 37   this.schema = null;
 38 };
 39 
 40 module.exports = API;

 50 /**
 51  * Discovers urls from Spire API.  Since this description does not change often,
 52  * we only make the request once and cache the result for subsequent calls.
 53  *
 54  * @example
 55  * api.discover(function (err, discovered) {
 56  *   if (!err) {
 57  *     // ...
 58  *   }
 59  * });
 60  *
 61  * @param {function(err, discovered)} cb Callback
 62  */
 63 API.prototype.discover = function (cb) {
 64   var api = this;
 65 
 66   if (this.description) {
 67     return cb(null, this.description);
 68   }
 69 
 70   this.request('discover', function (err, description) {
 71     if (err) return cb(err);
 72     api.description = description;
 73     api.schema = description.schema[api.version];
 74     cb(null, description);
 75   });
 76 };

 78 /**
 79  * Creates a spire session from an account key.
 80  *
 81  * @param {string} key The acccount key
 82  * @param {function(err)} cb Callback
 83  */
 84 API.prototype.createSession = function (key, cb) {
 85   var api = this;
 86   this.request('create_session', key, function (err, sessionData) {
 87     if (err) return cb(err);
 88     session = new Session(api.spire, sessionData);
 89     cb(null, session);
 90   });
 91 };

152 /**
153  * Returns the MIME type for resourceName.
154  *
155  * @param {string} [name] Name of the resource MIME type to return
156  * @returns {string} MIME type of the resource
157  */
158 API.prototype.mediaType = function(resourceName){
159   return this.schema[resourceName].mediaType;
160 };
161 
162 /**
163  * Returns the Authorization header for the resource.
164  *
165  * @param {Resource} resource Resource
166  * @returns {string} Authorization header for the resource
167  */
168 API.prototype.authorization = function(resource){
169   return ['Capability', resource.capability].join(' ');
170 };

178 /**
179  * Gets the api description resource.
180  * @name discover
181  * @ignore
182  */
183 Resource.defineRequest(API.prototype, 'discover', function () {
184   return {
185     method: 'get',
186     url: this.url,
187     headers: {
188       accept: "application/json"

190   };

193 /**
194  * Posts to sessions url with the accont key.
195  * @name create_session
196  * @ignore
197  */
198 Resource.defineRequest(API.prototype, 'create_session', function (key) {
199   var spire = this.spire;
200   return {
201     method: 'post',
202     url: this.description.resources.sessions.url,
203     headers: {
204       'Content-Type': this.mediaType('account'),
205       'Accept': this.mediaType('session')
206     },
207     content: {key: key}
208   }
209 });
