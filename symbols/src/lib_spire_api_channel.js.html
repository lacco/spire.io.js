1 /**
  2  * @fileOverview Channel Resource class definition
  3  */
  4 var Resource = require('./resource');
  5 
  6 /**
  7  * Represents a channel in the spire api.
  8  *
  9  * @class Channel Resource
 10  *
 11  * @constructor
 12  * @extends Resource
 13  * @param {object} spire Spire object
 14  * @param {object} data  Channel data from the spire api
 15  */
 16 function Channel (spire, data) {
 17   this.spire = spire;
 18   this.data = data;
 19   this.resourceName = 'channel';
 20  21 };
 22 
 23 Channel.prototype = new Resource();
 24 
 25 module.exports = Channel;
 26 
 27 /**
 28  * Returns the channel name.
 29  *
 30  * @returns {string} Channel name
 31  */
 32 Channel.prototype.name = function () {
 33   return this.data.name;
 34 };
 35 
 36 /**
 37  * Publishes a message to the channel.
 38  *
 39  * The messages can be a string, or any json'able object.
 40  *
 41  * @example
 42  * spire.channel('myChannel', function (err, channel) {
 43  *   channel.publish('hello world', function (err, message) {
 44  *     if (!err) {
 45  *       // Message has been published.
 46  *     }
 47  *   });
 48  * });
 49  *
 50  * @param {string|object} message Message to publish
 51  * @param {function (err, message)} cb Callback
 52  */
 53 Channel.prototype.publish = function (message, cb) {
 54   this.request('publish', { content: message }, cb);
 55 };
 56 
 57 /**
 58  * Gets a subscription to a channel, creating it if necessary.
 59  *
 60  * @example
 61  * spire.channel('myChannel', function (err, channel) {
 62  *   channel.subscription('mySubscription', function (err, subscription) {
 63  *     if (!err) {
 64  *       // `subscription` is the new subscription.
 65  *     }
 66  *   });
 67  * });
 68  *
 69  * @param {string} subName Subscription name
 70  * @param {function (err, subscription)} cb Callback
 71  */
 72 Channel.prototype.subscription = function (subName, cb) {
 73   if (!cb) {
 74     cb = subName;
 75     name = null;
 76   }
 77   this.spire.subscription(subName, this.name(), cb);
 78 };
 79 
 80 /**
 81  * Requests
 82  *
 83  * These define API calls and have no side effects.  They can be run by calling
 84  *     this.request(<request name>);
 85  */
 86 
 87 /**
 88  * @name publish
 89  * @ignore
 90  * Publishes a message to the channel.
 91  */
 92 Resource.defineRequest(Channel.prototype, 'publish', function (message) {
 93   var spire = this.spire;
 94   return {
 95     method: 'post',
 96     url: this.url(),
 97     headers: {
 98       'Authorization': this.authorization(),
 99       'Accept': this.mediaType('message'),
100       'Content-Type': this.mediaType('message')
101     },
102     content: message,
103   };
104 });
