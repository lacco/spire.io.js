1 /**
  2  * @fileOverview Resource class definition
  3  */
  4 var Resource = require('./resource');
  5 
  6 /**
  7  * Represents a channel in the spire api.
  8  *
  9  * @constructor
 10  * @extends Resource
 11  * @param {object} spire Spire object
 12  * @param {object} data  Channel data from the spire api
 13  */
 14 function Channel (spire, data) {
 15   this.spire = spire;
 16   this.data = data;
 17   this.resourceName = 'channel';
 18 };
 19 
 20 Channel.prototype = new Resource();
 21 
 22 module.exports = Channel;
 23 
 24 /**
 25  * Returns the channel name.
 26  *
 27  * @returns {string} Channel name
 28  */
 29 Channel.prototype.name = function () {
 30   return this.data.name;
 31 };
 32 
 33 /**
 34  * Publishes a message to the channel.
 35  *
 36  * @example
 37  * spire.channel('myChannel', function (err, channel) {
 38  *   channel.publish('hello world', function (err, message) {
 39  *     if (!err) {
 40  *       // Message has been published.
 41  *     }
 42  *   });
 43  * });
 44  *
 45  * @param {string} message Message to publish
 46  * @param {function (err, message)} cb Callback
 47  */
 48 Channel.prototype.publish = function (message, cb) {
 49   this.request('publish', { content: message }, cb);
 50 };
 51 
 52 /**
 53  * Gets a subscription to a channel, creating it if necessary.
 54  *
 55  * @example
 56  * spire.channel('myChannel', function (err, channel) {
 57  *   channel.subscribe('mySubscription', function (err, subscription) {
 58  *     if (!err) {
 59  *       // `subscription` is the new subscription.
 60  *     }
 61  *   });
 62  * });
 63  *
 64  * @param {string} subName Subscription name
 65  * @param {function (err, subscription)} cb Callback
 66  */
 67 Channel.prototype.subscribe = function (subName, cb) {
 68   if (!cb) {
 69     cb = subName;
 70     name = null;
 71   }
 72   this.spire.subscribe(subName, this.name(), cb);
 73 };
 74 
 75 /**
 76  * Requests
 77  *
 78  * These define API calls and have no side effects.  They can be run by calling
 79  *     this.request(<request name>);
 80  */
 81 
 82 /**
 83  * @name publish
 84  * @ignore
 85  * Publishes a message to the channel.
 86  */
 87 Resource.defineRequest(Channel.prototype, 'publish', function (message) {
 88   var spire = this.spire;
 89   return {
 90     method: 'post',
 91     url: this.url(),
 92     headers: {
 93       'Authorization': this.authorization(),
 94       'Accept': this.mediaType('message'),
 95       'Content-Type': this.mediaType('message')
 96     },
 97     content: message,
 98   };
 99 });
