  5 var Resource = require('./resource')
  6   , _ = require('underscore')
  7   , async = require('async')

 10 /**
 11  * Represents a subscription in the spire api.
 12  *
 13  * <p>There are a few ways to get messages from a subscription.
 14  *
 15  * <p>The first is to call <code>subscription.retrieveMessages</code> directly.
 16  * This is the most general method, and supports a number of options.
 17  *
 18  * <p>There are convenience methods <code>subscription.poll</code and
 19  * <code>subscription.longPoll</code> which wrap <code>retrieveMessages</code>.
 20  * The only difference is that <code>subscription.poll</code> has a timeout of
 21  * 0, so the request will always come back right away, while
 22  * <code>subscription.longPoll</code> has a 30 second timeout, so the request
 23  * will wait up to 30 seconds for new messages to arrive before returning.
 24  *
 25  * <p>You can also use the <code>message</code> and <code>messages</code> events to
 26  * listen for new messages on the subscription.
 27  *
 28  * <p><pre><code>
 29  *    subscription.addListener('message', function (message) {
 30  *      console.log('Message received: ' + message.content);
 31  *    });
 32  *
 33  34  *    subscription.addListener('messages', function (messages) {
 35  *      console.log('Received ' + messages.length + ' messages.');
 36  *    });
 37  *
 38  *    subscription.startListening();
 39  * </code></pre>
 40  * </p>
 41  *
 42  * <p>The `messages` event fires first and contains the all the messages that were
 43  * received in a single request.  The `message` event fires once per message.
 44  *
 45  * @class Subscription Resource
 46  *
 47  * @constructor
 48  * @extends Resource
 49  * @param {object} spire Spire object
 50  * @param {object} data Subscription data from the spire api
 51  */
 52 function Subscription(spire, data) {
 53   this.spire = spire;
 54   this.data = data;
 55   this.resourceName = 'subscription';
 56 
 57   this.last = null;
 58   this.listening = false;
 59 }
 60 
 61 Subscription.prototype = new Resource();
 62 
 63 module.exports = Subscription;
 64 
 65 /**
 66  * Gets the name of the subscription.
 67  *
 68  * @returns {string} Name
 69  */
 70 Subscription.prototype.name = function () {
 71   return this.data.name;
 72 };

 74 /**
 75  * Starts long polling for the subscription.
 76  *
 77  * <p>The <code>message</code> and <code>messages</code> events will fire when a
 78  * request comes back with messages.  The <code>message</code> event will fire
 79  * once per message, while the <code>messages</code> event fires every time a
 80  * request comes back with more than one message.
 81  *
 82  * @example
 83  * subscription.addListener('message', function (message) {
 84  *   console.log('Message received: ' + message.content);
 85  * });
 86  *
 87  * subscription.addListener('messages', function (messages) {
 88  *   console.log('Received ' + messages.length + ' messages.');
 89  * });
 90  *
 91  * subscription.startListening();
 92  *
 93  * // Stop Listening after 100 seconds.
 94  * setTimout(function () {
 95  *   subscription.stopListening();
 96  *  }, 100000);
 97  *
 98  * @param {object} [options] Optional options argument
 99  * @param {number} [options.last] Optional last message
100  * @param {number} [options.delay] Optional delay
101  * @param {number} [options.timeout] Optional timeout
102  * @param {string} [options.orderBy] Optional ordering ('asc' or 'desc')
103  * @param {function (err, messages)} cb Callback
104  */
105 Subscription.prototype.startListening = function (opts) {
106   this.listening = true;
107   this._listen(opts);
108 };
109 
110 /**
111  * Stops listening on the subscription.
112  */
113 Subscription.prototype.stopListening = function () {
114   this.listening = false;
115 };
116 
117 /**
118  * Gets messages for the subscription.
119  *
120  * <p>This method only makes one request.  Use
121  * <code>subscription.startListening</code> to poll repeatedly.
122  *
123  * @example
124  * subscription.retrieveMessages(function (err, messages) {
125  *   if (!err) {
126  *     // `messages` is an array of messages (possably empty)
127  *   }
128  * });
129  *
130  * @param {object} [options] Optional options argument
131  * @param {number} [options.last] Optional last message
132  * @param {number} [options.delay] Optional delay
133  * @param {number} [options.timeout] Optional timeout
134  * @param {string} [options.orderBy] Optional ordering ('asc' or 'desc')
135 136  * @param {function (err, messages)} cb Callback
137  */
138 Subscription.prototype.retrieveMessages = function (options, cb) {
139   var subscription = this;
140   if (!cb) {
141     cb = options;
142     options = {};
143   }
144 
145   options.orderBy = options.orderBy || 'desc';
146 
147   this.request('messages', options, function (err, messagesData) {
148     if (err) {
149       subscription.emit('error', err);
150       return cb(err);
151     }
152 
153     var messages = messagesData.messages;
154 
155     if (messages.length && subscription.listening) {
156       subscription.emit('messages', messages);
157       _.each(messages, function (message) {
158         subscription.emit('message', message);
159       });
160     }
161 
162     cb(null, messages);
163   });
164 };
165 
166 /**
167  * Alias for subscription.retreiveMessages.
168  */
169 Subscription.prototype.get = Subscription.prototype.retreiveMessages;
170 
171 /**
172  * Gets new messages for the subscription.  This method forces a 0 second
173  * timeout, so the request will come back immediately, but may have an empty
174  * array of messages if there are no new ones.
175  *
176  * <p>This method only makes one request.  Use
177  * <code>subscription.startListening</code> to poll repeatedly.
178  *
179  * @example
180  * subscription.poll(function (err, messages) {
181  *   if (!err) {
182  *     // `messages` is an array of messages (possably empty)
183  *   }
184  * });
185  *
186  * @param {object} [options] Optional options argument
187  * @param {number} [options.delay] Optional delay
188  * @param {string} [options.orderBy] Optional ordering ('asc' or 'desc')
189  * @param {function (err, messages)} cb Callback
190  */
191 Subscription.prototype.poll = function (options, cb) {
192   if (!cb) {
193     cb = options;
194     options = {};
195   }
196   options.timeout = 0;
197   this.longPoll(options, cb);
198 };
199 
200 /**
201  * Gets new messages for the subscription.
202  *
203  * <p>This method defaults to a 30 second timeout, so the request will wait up to
204  * 30 seconds for a new message to come in.  You can increase the wait time with
205  * the <code>options.timeout</code> paraameter.
206  *
207  * <p>This method only makes one request.  Use `subscription.startListening` to
208  * poll repeatedly.
209  *
210  * @example
211  * subscription.longPoll({ timeout: 60 }, function (err, messages) {
212  *   if (!err) {
213  *     // `messages` is an array of messages (possably empty)
214  *   }
215  * });
216  *
217  * @param {object} [options] Optional options argument
218  * @param {number} [options.delay] Optional delay
219  * @param {string} [options.orderBy] Optional ordering ('asc' or 'desc')
220  * @param {number} [options.timeout] Optional timeout
221  * @param {function (err, messages)} cb Callback
222  */
223 Subscription.prototype.longPoll = function (options, cb) {
224   var subscription = this;
225   if (!cb) {
226     cb = options;
227     options = {};
228   }
229 
230   // timeout option of 0 means no long poll,
231   // so we force it here.
232   options.last = this.last;
233   options.timeout = options.timeout || 30;
234   this.retrieveMessages(options, function (err, messages) {
235     if (err) return cb(err);
236     if (messages.length) {
237       if (options.orderBy === 'asc') {
238         subscription.last = _.first(messages).timestamp;
239       } else {
240         subscription.last = _.last(messages).timestamp;
241       }
242     }
243     cb(null, messages);
244   });
245 };
246 
247 /**
248  * Repeatedly polls for new messages until `subscription.stopListening` is
249  * called.
250  *
251  * You should use `subscription.startListening` instead of calling this method
252  * directly.
253  */
254 Subscription.prototype._listen = function (opts) {
255   var subscription = this;
256   opts = opts || {};
257   async.whilst(
258     function () { return subscription.listening; },
259     function (cb) {
260       subscription.longPoll(opts, cb);
261     },
262     function () {}
263   );
264 };

273 /**
274  * Gets the messages for the subscription, according to various parameters.
275  * @name messages
276  * @ignore
277  */
278 Resource.defineRequest(Subscription.prototype, 'messages', function (options) {
279   options = options || {};
280   return {
281     method: 'get',
282     url: this.url(),
283     query: {
284       'timeout': options.timeout || 0,
285       'last-message': options.last || 0,
286       'order-by': options.orderBy || 'desc',
287       'delay': options.delay || 0
288     },
289     headers: {
290       'Authorization': this.authorization(),
291       'Accept': this.mediaType('events')
292     }
293   };
294 });
